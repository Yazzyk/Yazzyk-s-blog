import {INPUT_VALUE_CHANGE,ADD_TODO_ITEM,DEL_TODO_ITEM} from './ActionTypes';

export const getInputChangeAction = (value) =>({
    type : INPUT_VALUE_CHANGE,
    value
});

export const getAddItemAction = () =>({
    type: ADD_TODO_ITEM
});

export const getDelItemAction = (index)=>({
    type: DEL_TODO_ITEM,
    index
});
